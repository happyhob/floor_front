# React 앱 생성
npx create-react-app client

# axios 라이브러리 설치
npm install axios

# router 라이브러리 설치
npm install react-router-dom

# sweetalert2 라이브러리 설치
npm install sweetalert2
npm install sweetalert2-react-content

# 웹 프로젝트에서 사용할 수 있는 무료 솔리드 아이콘 모음을 제공
npm install @fortawesome/free-solid-svg-icons

# React 애플리케이션에서 Font Awesome 아이콘을 쉽게 사용
npm install @fortawesome/react-fontawesome

# React 애플리케이션에서 Material-UI 라이브러리와 함께 사용할 수 있는 머티리얼 디자인 아이콘을 제공
npm install @mui/icons-material

# Drei는 웹용 인기 3D 라이브러리인 three.js에 대한 유용한 도우미 및 추상화 세트
npm install @react-three/drei

# React Three Fiber는 Three.js용 React 렌더러로, React 애플리케이션에서 선언적 3D 렌더링을 가능하게 합니다.
npm install @react-three/fiber

# 이 라이브러리는 테스트 라이브러리로 DOM 요소를 테스트하기 위한 사용자 정의 Jest 매처를 제공
npm install @testing-library/jest-dom

# 이 라이브러리는 간단하고 직관적인 API를 사용하여 React 구성 요소를 테스트하기 위한 유틸리티를 제공
npm install @testing-library/react

# 이 라이브러리는 React 구성 요소를 테스트할 때 사용자 이벤트(예: 클릭, 입력)를 시뮬레이션하기 위한 유틸리티를 제공
npm install @testing-library/user-event

# 이 패키지는 three.js 라이브러리에 대한 TypeScript 유형 정의를 제공하여 TypeScript 프로젝트에서 three.js를 사용한 유형 안전 개발을 가능
npm install @types/three

#  Anime.js는 애니메이션 제작을 위한 간단하고 강력한 API를 갖춘 경량 JavaScript 애니메이션 라이브러리
npm install animejs

# Axios는 브라우저 및 Node.js용으로 널리 사용되는 Promise 기반 HTTP 클라이언트로, HTTP 요청을 쉽게 만들 수 있습니다.
npm install axios


# Qrcode.react는 QR 코드 생성을 위한 React 구성 요소
npm install qrcode.react

# React Bootstrap은 Bootstrap을 사용하여 반응형 모바일 우선 웹 애플리케이션을 구축하기 위한 재사용 가능한 React 구성 요소 라이브러리
npm install react-bootstrap

#  이 라이브러리는 React 애플리케이션에서 장치 유형 및 브라우저와 같은 다양한 장치 속성을 감지하기 위한 유틸리티를 제공
npm install react-device-detect

# React DOM은 DOM에서 React의 진입점으로, 브라우저에서 React 구성 요소를 렌더링할 수 있습니다.
npm install react-dom

# React Draggable은 React 애플리케이션에서 드래그 가능한 요소를 활성화하는 React 구성 요소
npm install react-draggable

# React Icons는 React 애플리케이션에 쉽게 통합할 수 있도록 React 구성 요소로 제공되는 인기 있는 아이콘 팩 라이브러리
npm install react-icons

#  React Modal은 접근 가능한 모달 대화 상자를 만들기 위한 React 구성 요소
npm install react-modal

# React Router는 React 애플리케이션을 위한 널리 사용되는 라우팅 라이브러리로, 선언적 라우팅 및 탐색을 가능하게 합니다.
npm install react-router

# React Router DOM은 BrowserRouter, Link, Route 등을 포함한 웹 애플리케이션용 React Router에 대한 바인딩을 제공
npm install react-router-dom

# React Scripts는 React 애플리케이션을 구축하기 위해 Create React App에서 사용하는 스크립트와 구성을 제공
npm install react-scripts

#  React Three Fiber는 Three.js용 React 렌더러로, React 애플리케이션에서 선언적 3D 렌더링을 가능
npm install react-three-fiber

# 이 라이브러리는 웹 애플리케이션에서 사용할 수 있도록 Start Bootstrap의 간단한 사이드바 템플릿을 제공
npm install startbootstrap-simple-sidebar

# Three.js는 WebGL을 사용하여 브라우저에서 3D 그래픽을 생성하고 렌더링하는 데 널리 사용되는 JavaScript 라이브러리
npm install three

# Web Vitals는 웹 페이지의 주요 사용자 중심 성능 지표를 측정하기 위한 라이브러리
npm install web-vitals
